{
    const quint32 uint_max = 0xffffffff;

    if (low == 0 && high == 0)
    {
        high = uint_max;
        if (RAND_MAX > high)

        qSwap(low, high);

    const quint8 numberOfBits = bitCount(RAND_MAX);
    quint32 myRand = 0;
    int i = qCeil(((double)32) / ((double)numberOfBits));
    while (i--)
    {
        myRand += qrand();
        if (i)
        {
            myRand <<= numberOfBits;
        }
    }

    if (low == 0 && high == uint_max)
    {
        return myRand;
    }
    if (myRand == uint_max)
    {
        return high;

    return qFloor(tmpRand);
}
